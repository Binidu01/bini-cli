## Quick Start

```bash
npx create-bini-app@latest my-app
cd my-app
npm install
npm run dev
```

Opens `http://localhost:3000` automatically.

---

## What You Get

Running `create-bini-app` scaffolds a complete project with:

- **File-based routing** via `bini-router` тАФ `page.tsx` files map to URLs, nested layouts, per-route metadata, automatic code splitting
- **API routes** powered by Hono тАФ plain function handlers or full Hono apps in `src/app/api/`
- **Zero-dependency production server** via `bini-server` тАФ pure Node.js `http`, serves `dist/` and proxies `/api/*`
- **Dev overlay** via `bini-overlay` тАФ animated logo badge and full error overlay with source maps
- **Clean startup banner** via `bini-env` тАФ shows active `.env` files on dev server start
- **Automatic favicons** тАФ SVG, PNG at 16├Ч16 through 512├Ч512, `apple-touch-icon`, `og-image.png`, and `site.webmanifest` generated at scaffold time
- **Official plugin system** тАФ extend Bini.js with first-party and community plugins via `bini.config.mjs`
- **TypeScript or JavaScript** тАФ your choice at scaffold time
- **Tailwind CSS, CSS Modules, or plain CSS** тАФ your choice at scaffold time

## File-Based Routing

`bini-router` maps `page.tsx` files to URLs тАФ pure SPA, no server required at runtime.

```
src/app/
  page.tsx                 тЖТ /
  dashboard/
    layout.tsx             тЖТ wraps /dashboard and all children
    page.tsx               тЖТ /dashboard
    [id]/
      page.tsx             тЖТ /dashboard/:id
  blog/
    [slug]/
      page.tsx             тЖТ /blog/:slug
  not-found.tsx            тЖТ custom 404 (optional)
```

### Pages

```tsx
export default function Dashboard() {
  return <h1>Dashboard</h1>
}
```

### Dynamic routes

```tsx
// src/app/blog/[slug]/page.tsx
import { useParams } from 'react-router-dom'

export default function Post() {
  const { slug } = useParams()
  return <h1>Post: {slug}</h1>
}
```

### Layouts

Root layouts use `{children}` тАФ they wrap `<BrowserRouter>` from outside.
Nested layouts use `<Outlet />` тАФ they are React Router route wrappers.

```tsx
// src/app/layout.tsx тАФ root layout
export const metadata = {
  title      : 'My App',
  description: 'Built with Bini.js',
}

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return <>{children}</>
}
```

```tsx
// src/app/dashboard/layout.tsx тАФ nested layout
import { Outlet } from 'react-router-dom'

export default function DashboardLayout() {
  return (
    <div>
      <aside>Sidebar</aside>
      <main><Outlet /></main>
    </div>
  )
}
```

### Metadata

Export `metadata` from any `layout.tsx`. All fields are optional. Metadata is stripped from the browser bundle at build time тАФ it never ships to the client.

```tsx
export const metadata = {
  title      : 'Dashboard',
  description: 'Your personal dashboard',
  themeColor : '#00CFFF',
  manifest   : '/site.webmanifest',
  openGraph  : {
    title : 'Dashboard',
    url   : 'https://myapp.com/dashboard',
    images: [{ url: '/og-image.png', width: 1200, height: 630 }],
  },
  twitter: {
    card   : 'summary_large_image',
    creator: '@yourhandle',
    images : ['/og-image.png'],
  },
  icons: {
    icon : [{ url: '/favicon.svg', type: 'image/svg+xml' }],
    apple: [{ url: '/apple-touch-icon.png', sizes: '180x180' }],
  },
}
```

## API Routes

Create files in `src/app/api/`. Both **plain function handlers** and **Hono apps** are supported. The same files run in dev and production.

```
src/app/api/
  hello.ts           тЖТ /api/hello
  users.ts           тЖТ /api/users
  posts/
    index.ts         тЖТ /api/posts
    [id].ts          тЖТ /api/posts/:id
  [...catch].ts      тЖТ /api/* catch-all
```

### Plain function handler

```ts
// src/app/api/hello.ts

// Return a plain object тАФ auto-wrapped as JSON
export default function handler(req: Request) {
  return { message: 'hello', method: req.method }
}

// Or return an explicit Response
export default function handler(req: Request) {
  return Response.json({ message: 'hello' })
}
```

### Hono app

```ts
// src/app/api/users.ts
import { Hono } from 'hono'

const app = new Hono().basePath('/api')

app.get('/users', (c) => c.json({ users: ['alice', 'bob'] }))

app.post('/users', async (c) => {
  const body = await c.req.json()
  return c.json({ created: body }, 201)
})

export default app
```

### Dynamic API routes

```ts
// src/app/api/posts/[id].ts
import { Hono } from 'hono'

const app = new Hono().basePath('/api')

app.get('/posts/:id', (c) => c.json({ id: c.req.param('id') }))

export default app
```

## Plugin System

Bini.js now has an **official plugin system**. Plugins let you extend the framework with reusable behaviour тАФ hook into the dev server, build pipeline, routing, and API layer without touching your app code.

### Using plugins

Install a plugin and register it in `bini.config.mjs`:

```js
// bini.config.mjs
import { defineConfig } from 'bini-router'
import myPlugin from 'bini-plugin-example'

export default defineConfig({
  plugins: [
    myPlugin({ /* options */ }),
  ],
})
```

Multiple plugins are supported and run in order:

```js
export default defineConfig({
  plugins: [
    authPlugin(),
    analyticsPlugin({ trackPageViews: true }),
    i18nPlugin({ locales: ['en', 'si'] }),
  ],
})
```

### Writing a plugin

A Bini.js plugin is a function that returns a plugin object:

```ts
import type { BiniPlugin } from 'bini-router'

export default function myPlugin(options = {}): BiniPlugin {
  return {
    name: 'bini-plugin-my-plugin',

    // Called once when the dev server starts
    onDevStart({ server }) {
      console.log('Dev server ready')
    },

    // Called before every build
    onBuildStart({ config }) {
      console.log('Building...')
    },

    // Called after every build
    onBuildEnd({ outDir }) {
      console.log(`Built to ${outDir}`)
    },

    // Extend the Hono API router
    onApiSetup({ app }) {
      app.get('/api/my-plugin/status', (c) => c.json({ ok: true }))
    },
  }
}
```

All lifecycle hooks are optional тАФ implement only what your plugin needs.

### Available hooks

| Hook | When it runs | Arguments |
|---|---|---|
| `onDevStart` | Dev server ready | `{ server, config }` |
| `onBuildStart` | Before `vite build` | `{ config }` |
| `onBuildEnd` | After `vite build` | `{ outDir, config }` |
| `onApiSetup` | API router initialised | `{ app: Hono }` |
| `onRequest` | Every incoming request | `{ req, next }` |

### Official plugins

| Plugin | Description |
|---|---|
| [`bini-plugin-auth`](https://npmjs.com/package/bini-plugin-auth) | Session-based and JWT authentication |
| [`bini-plugin-i18n`](https://npmjs.com/package/bini-plugin-i18n) | Internationalisation and locale routing |
| [`bini-plugin-analytics`](https://npmjs.com/package/bini-plugin-analytics) | Privacy-first page view analytics |

> Community plugins follow the `bini-plugin-*` naming convention. See the [plugin registry](https://bini.js.org/plugins) for the full list.

## Production Server

`bini-server` is a **zero-dependency** production server тАФ pure Node.js `http`, no Express, no Fastify.

```bash
npm run build   # vite build тЖТ dist/
npm start       # bini-server
```

**Terminal output:**
```
  ├Я Bini.js (production)

  тЮЬ  Environments: .env, .env.production
  тЮЬ  Local:   http://localhost:3000/
  тЮЬ  Network: http://192.168.1.5:3000/
```

**Request flow:**
```
Request
  тФЬтФА /api/*  тЖТ  src/app/api/ handlers  (Hono apps or plain functions)
  тФЬтФА /*      тЖТ  stream static file from dist/
  тФФтФА /*      тЖТ  dist/index.html  (SPA fallback)
```

| Feature | `vite preview` | `bini-server` |
|---|---|---|
| Serves `dist/` | тЬЕ | тЬЕ |
| API routes | тЬЕ | тЬЕ |
| SPA fallback | тЬЕ | тЬЕ |
| Production use | тЭМ not recommended | тЬЕ |
| Body timeout | тЭМ | тЬЕ 30s |
| Body size limit | тЭМ | тЬЕ 10 MB |
| Handler timeout | тЭМ | тЬЕ 30s |
| Graceful shutdown | тЭМ | тЬЕ |
| Zero dependencies | тЬЕ | тЬЕ |

Override the default port `3000` via environment variable:

```bash
PORT=8080 npm start
```

## Deployment

### Node.js servers (Railway, Render, Fly.io, VPS)

```bash
npm run build
npm start
```

Railway and Render inject `PORT` automatically. Use [pm2](https://pm2.keymetrics.io/) on a VPS:

```bash
npm install -g pm2
pm2 start "npm start" --name my-app
pm2 save && pm2 startup
```

### Netlify (edge functions)

```js
// vite.config.mjs
biniroute({ platform: 'netlify' })
```

```toml
# netlify.toml
[build]
  command = "vite build"
  publish = "dist"

[[edge_functions]]
  path     = "/api/*"
  function = "api"

[[redirects]]
  from   = "/*"
  to     = "/index.html"
  status = 200
```

> Use `[[edge_functions]]` for the API тАФ not `[[redirects]]`.

### Cloudflare Workers

```bash
npm install wrangler
```

```js
biniroute({ platform: 'cloudflare' })
```

```bash
vite build && npx wrangler deploy
```

### Other platforms

`bini-router` supports `netlify` ┬╖ `cloudflare` ┬╖ `node` ┬╖ `deno` ┬╖ `bun` ┬╖ `aws`. Set `platform` in `vite.config.mjs` once тАФ `vite build` generates the platform entry file automatically.

> **Static hosting (GitHub Pages, S3, Netlify static) is not supported.** API routes require a Node.js runtime.

## Environment Variables

`bini-env` detects and displays active `.env` files on every server start (dev and preview):

```
  ├Я Bini.js (dev)

  тЮЬ  Environments: .env.local, .env
  тЮЬ  Local:   http://localhost:3000/
  тЮЬ  Network: http://192.168.1.10:3000/
```

Files are checked in priority order: `.env.local` тЖТ `.env.[mode].local` тЖТ `.env.[mode]` тЖТ `.env`

Prefix variables with `VITE_` to expose them to the browser:

```bash
# .env
VITE_API_URL=https://api.example.com

# .env.local (overrides .env, git-ignored)
VITE_API_URL=http://localhost:4000
```

## Dev Overlay

`bini-overlay` adds an animated Bini.js logo badge to the bottom-left corner during development. It has three states:

| State | Behaviour |
|---|---|
| Loading | Logo draws itself with a stroke animation and pulses |
| Idle | Logo sits quietly as a filled gradient icon |
| Error | Badge hides тАФ full error overlay with source maps takes over |

The badge recovers automatically when an error is fixed and HMR fires. **Never appears in production builds.**

```bash
# Disable without touching config
DISABLE_BINI_OVERLAY=true npm run dev
```
